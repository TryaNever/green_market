pm.test("Test Status 200", function() {
    pm.response.to.have.status(200)
})

pm.test("response must be valid and have a body", function () {
     pm.response.to.be.ok;
     pm.response.to.be.withBody;
     pm.response.to.be.json;
});

pm.test("Response time is less than 800ms", () => {
  pm.expect(pm.response.responseTime).to.be.below(800);
  pm.response.to.have.responseTime.not.above(800);
  pm.expect(pm.response.responseTime).to.be.below(800);
});

pm.test("Body contains string",() => {
  pm.expect(pm.response.text()).to.include("id");
  pm.expect(pm.response.text()).to.include("name");
  pm.expect(pm.response.text()).to.include("description");
  pm.expect(pm.response.text()).to.include("unitPrice");
  pm.expect(pm.response.text()).to.include("stock");
  pm.expect(pm.response.text()).to.include("seller");
  pm.expect(pm.response.text()).to.include("createdAt");
});

pm.test("Content-Type header is present", () => {
  pm.response.to.have.header("Content-Type");
  pm.test("Content-Type header is application/json", () => {
  pm.expect(pm.response.headers.get('Content-Type')).to.include('application/json');
});
});
